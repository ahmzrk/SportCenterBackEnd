version: "3.9"

services:
  # RabbitMQ servisi
  rabbitmq:
    image: rabbitmq:3-management-alpine
    container_name: rabbitmq
    ports:
      - "5672:5672"
      - "15672:15672"
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "check_port_connectivity"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Payment servisi
  payment:
    build:
      context: .
      dockerfile: PaymentService/WebAPI/Dockerfile
    depends_on:
      rabbitmq:
        condition: service_healthy # RabbitMQ hazır olana kadar bekle

  # PDFReport servisi
  pdfreport:
    build:
      context: .
      dockerfile: PDFReportService/API/Dockerfile
    depends_on:
      rabbitmq:
        condition: service_healthy # RabbitMQ hazır olana kadar bekle

  # SportsCenter servisi
  sportscenter:
    build:
      context: .
      dockerfile: SportsCenter/WebAPI/Dockerfile
    depends_on:
      rabbitmq:
        condition: service_healthy # RabbitMQ hazır olana kadar bekle

  # APIGateway servisi
  apigateway:
    build:
      context: .
      dockerfile: APIGetaway/APIGetaway/Dockerfile
    ports:
      - "7084:80" # Dışarıya 8084, içeriye 8080 portunu açar.
    depends_on:
      - sportscenter
      - pdfreport
      - payment
